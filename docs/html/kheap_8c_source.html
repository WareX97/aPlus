55 static int bitmap_first(heap_t* heap, size_t size) {
56  if(size == 0)
57  return -1;
58 
59  for(int i = 0; i < heap->size; i++) {
60  for(int j = 0, f = 0; j < size; j++) {

66  return i;
67  }
68  }

70  return -1;
71 }
