  29 /*
  30  * 0x00000000 - 0x00800000:             Kernel Reserved
  31  * 0x00800000 - 0x00A00000:             Not available
  32  * 0x00A00000 - 0x40000000:     User executable
  33  * 0x40000000 - 0xE0000000:             Kernel Heap
  34  * 0xE0000000 - 0xF0000000;             Linear Frame Buffer (User)
  35  * 0xF0000000 - 0xFFFFFFFF;             Kernel Heap

  71 static inline void* mm_paddr(void* vaddr) {
  72         if((uint32_t) vaddr > MM_VBASE)
  73                 vaddr = (void*) ((uint32_t) vaddr - MM_VBASE);
  74                 
  75         return vaddr;
  76 }
  77 
  78 static inline void* mm_vaddr(void* paddr) {
  79         if((uint32_t) paddr < MM_VBASE)
  80                 paddr = (void*) ((uint32_t) paddr + MM_VBASE);
  81                 
  82         return paddr;
  83 }

  85 static inline void* mm_align(void* vaddr) {
  86         return (void*) ((uint32_t) vaddr & ~0xFFF);
