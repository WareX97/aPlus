  23 #ifndef _SPINLOCK_H
  24 #define _SPINLOCK_H
  25 
  26 #define SPINLOCK_FLAGS_LOCKED                                           1
  27 #define SPINLOCK_FLAGS_FASTLOCK                                         2
  28 
  29 typedef volatile int spinlock_t;
  30 
  31 void spinlock_lock(spinlock_t* spin);
  32 void spinlock_unlock(spinlock_t* spin);
  33 int spinlock_trylock(spinlock_t* spin); 
  34 void __spinlock_waiton();
  35 void __fastlock_waiton();
  36 
  37 
  38 #define lock()                                                                          \
  39         static spinlock_t __func__##_lock = 0;                  \
  40         spinlock_lock(&__func__##_lock)
  41         
  42 #define unlock()                                                                        \
  43         spinlock_unlock(&__func__##_lock)
  44         
  45         
  46 
  47 #define spinlock_waiton(cond)                                           \
  48         while(cond)                                                                             \
  49                 __spinlock_waiton()
  50 
  51 #define fastlock_waiton(cond)                                           \
  52         while(cond)                                                                             \
  53                 __fastlock_waiton()
  54 
  55 
  56 #define spinlock_init(spin, flags)                                      \
  57         *(spin) = flags
