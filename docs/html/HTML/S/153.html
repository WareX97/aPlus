  32 #define BITMAP_SET(bmp, bit)    \
  33         bmp[bit / 32] |= (1 << (bit % 32))
  34         
  35 #define BITMAP_CLR(bmp, bit)    \
  36         bmp[bit / 32] &= ~(1 << (bit % 32))

  38 #define BITMAP_TST(bmp, bit)    \
  39         (bmp[bit / 32] & (1 << (bit % 32)))

  52 static int bitmap_first(heap_t* heap, size_t size) {
  53         if(size == 0)
  54                 return -1;
  55                 
  56         for(int i = 0; i < heap->size; i++) {   
  57                 for(int j = 0, f = 0; j < size; j++) {
  58                         if(BITMAP_TST(heap->bitmap, (i + j)))
  59                                 continue;
  60                         
  61                         f++;    
  62                         if(f == size)
  63                                 return i;
  64                 }
  65         }
  66         
  67         return -1;
  68 }

  72 void* bitmap_alloc(heap_t* heap, size_t size) {
  73         if(!heap)
  74                 return 0;
  75                 
  76         if(!heap->bitmap)
  77                 return 0;
  78                 
  79         if(heap->used >= heap->size)
  80                 return 0;
  81                 
  82         
  83         if(size % BLKSIZE) {
  84                 size /= BLKSIZE;
  85                 size += 1;
  86         }else {
  87                 size /= BLKSIZE;
  88         }       
  89                 
  90         int index = bitmap_first(heap, size);
  91         if(index == -1)
  92                 return 0;
  93                                 
  94         for(int i = 0; i < size; i++)
  95                 BITMAP_SET(heap->bitmap, (index + i));
  96         
  97         
  98         heap->used += size;
  99         
 100         return (void*) (index * BLKSIZE);
 101 }

 103 void bitmap_free(heap_t* heap, void* addr, size_t size) {
 104         if(!heap)
 105                 return;
 106                 
 107         if(!heap->bitmap)
 108                 return;
 109                 
 110                 
 111         if(size % BLKSIZE) {
 112                 size /= BLKSIZE;
 113                 size += 1;
 114         }else {
 115                 size /= BLKSIZE;
 116         }       
 117                 
 118         
 119         int index = GETBIT(addr);
 120         for(int i = 0; i < size; i++)
 121                 BITMAP_CLR(heap->bitmap, (index + i));
 122                 
 123                 
 124         heap->used -= size;
 125 }

 129 int kheap_init() {
 130         extern uint32_t end;
 131 
 132         kheap.bitmap = &end;
 133         kheap.size = memsize / BLKSIZE;
 134         kheap.alloc = bitmap_alloc;
 135         kheap.free = bitmap_free;
 136         
 137         memset(kheap.bitmap, 0, kheap.size);
 138         
 139         mm_setheap(&kheap);
 140         
 141         // Alloc first 4MB (reserved physical kernel area)
 142         halloc(&kheap, (size_t) 0x400000);
 143         
 144 
 145         return 0;
 146 }
