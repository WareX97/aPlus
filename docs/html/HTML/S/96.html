  56 static inline int list_add(list_t* list, listval_t v) {
  57         list_body_t* val = (list_body_t*) kmalloc(sizeof(list_body_t));
  58         if(!val)
  59                 return -1;
  60                 
  61         spinlock_lock(&list->lock);
  62                 
  63         val->value = v;
  64         val->next = list->body;
  65         
  66         list->body = val;
  67         list->size += 1;
  68         
  69         spinlock_unlock(&list->lock);
  70         return 0;
  71 }

  73 static inline int list_remove(list_t* list, listval_t v) {
  74         spinlock_lock(&list->lock);
  75         
  76         list_body_t* body = list->body;
  77         list_body_t* prev = 0;
  78         
  79         while(body) {
  80                 if(body->value == v) {
  81                         if(prev)
  82                                 prev->next = body->next;
  83                         else
  84                                 list->body = body->next;
  85                                 
  86                         body->value = 0;
  87                         kfree(body);
  88                         break;
  89                 }
  90                 
  91                 prev = body;
  92                 body = body->next;
  93         }
  94         
  95         spinlock_unlock(&list->lock);
  96         return 0;
  97 }

  99 static inline int list_clear(list_t* list) {
 100         spinlock_lock(&list->lock);
 101         
 102         list_body_t* body = list->body;
 103         list_body_t* tmp = list->body;
 104         
 105         while(body) {
 106                 tmp = body->next;
 107                 kfree(body);
 108                 body = tmp;
 109         }

 119 static inline void list_clone(list_t* dest, list_t* src) {

 129 static inline listval_t list_prev(list_t* list, listval_t val) {
 130         for(list_body_t* i = list->body; i; i = i->next) {
 131                 if(i->value == val)
 132                         if(i->next)
 133                                 return i->next->value;
 134         }
 135         
 136         return (listval_t) NULL;
 137 }

 139 static inline listval_t list_next(list_t* list, listval_t val) {
 140         for(list_body_t* i = list->body; i; i = i->next) {
 141                 if(i->next)
 142                         if(i->next->value == val)
 143                                 return i->value;
 144         }
 145         
 146         return (listval_t) NULL;
 147 }
